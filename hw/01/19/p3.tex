\begin{problem}[5]
  Let $X$ be the set of all wffs of sentential logic
  and $\equiv$ be tautological equivalence.
  Define a binary (2-place) relation on equivalence classes by
  \[ [\alpha] \models [\beta] \iff \alpha \models \beta. \]
  Determine whether this relation is well-defined
  and prove your answer is correct.
\end{problem}
\begin{Answer}
  % \step
  % Let $[\alpha]$ be an equivalence class with the members
  % $\alpha_1, \alpha_2, \ldots, \alpha_n$,
  % and let $[\beta]$ be an equivalence class with the members
  % $\beta_1, \beta_2, \ldots, \beta_m$ for some $n, m \in \mathbb{Z}_{\ge 0}$.
  % By definition of equivalence classes,
  % $\alpha_i \equiv \alpha_j$ (so $\vbar(\alpha_i) = \vbar(\alpha_j))$ for all $i, j \in \{1, 2, \ldots, n\}$,
  % and $\beta_i \equiv \beta_j$ (so $\vbar(\beta_i) = \vbar(\beta_j))$ for all $i, j \in \{1,\ldots, m\}$.
  % This means that the value of $\vbar(\alpha \land \beta)$ is the same
  % for all $\alpha \in [\alpha]$ and for all $\beta \in [\beta]$.
  % Equivalently, $(\alpha_i \land \beta_k) \equiv (\alpha_j \land \beta_l)$
  % for all $\alpha_i, \alpha_j \in [\alpha]$ and for all $\beta_k, \beta_l \in [\beta]$.

  \step
  Let $v$ be a truth assignment.\\
  Suppose that $v(\alpha) = x \in \set{T, F}$, then $v(\alpha_i) = T$ for all $\alpha_i$ in the
  equivalence relation $[\alpha]$.\\
  Similarly, suppose that $v(\beta) = y \in \set{T, F}$, then $v(\beta_j) = T$ for all $\beta_j$ in the
  equivalence relation $[\beta]$.
  Then;
  \begin{align*}
    \vbar(\alpha_i \models \beta_j) = \vbar(\alpha \models \beta) = x \models y = \begin{cases}
      F & \text{ if } x = T \text{ and } y = F \\
      T & \text{ otherwise }
    \end{cases}
  \end{align*}
